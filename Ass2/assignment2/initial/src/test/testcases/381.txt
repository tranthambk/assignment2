
        int main()
        {
            int array[100], n, c, d, swap;
 
            printf("Enter number of elements\n");
            scanf("%d", n);
 
            printf("Enter %d integers\n", n);
 
            for (c = 0; c < n; c= c +1)
            scanf("%d", array[c]);
 
            for (c = 0 ; c < n - 1; c= c + 1)
            {
                for (d = 0 ; d < n - c - 1; d=d+1)
                {
                    if (array[d] > array[d+1]) /* For decreasing order use < */
                    {
                        swap       = array[d];
                        array[d]   = array[d+1];
                        array[d+1] = swap;
                    }
                }
            }
 
            printf("Sorted list in ascending order:\n");
 
            for (c = 0; c < n; c=c+1)
                 printf("%d\n", array[c]);
 
             return 0;
        }
