
        int fib(int n) 
        { 
            int a, b, c, i; a=0;b=1;
            if( n == 0) return a; 
            for (i = 2; i <= n; i = i + 1) 
            { 
                c = a + b; 
                a = b; 
                b = c; 
            } 
            return b; 
        }
