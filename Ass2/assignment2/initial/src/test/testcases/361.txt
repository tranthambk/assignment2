
        /* function to sort arr using shellSort */
        int shellSort(int arr[], int n) 
        {   
            int gap,i,j,temp;
            for (gap = n/2; gap > 0; gap = gap/ 2) 
            { 
                for (i = gap; i < n; i =i+ 1) 
                { 
                    temp = arr[i];          
                    for (j = i; j >= gap && arr[j - gap] > temp; j =j- gap) 
                        arr[j] = arr[j - gap]; 
                    arr[j] = temp; 
                } 
            } 
            return 0; 
        }
