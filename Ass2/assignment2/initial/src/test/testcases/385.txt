
        int giaiPT(float a, float b, float c,float x1, float x2){
            float delta ; delta= b*b - 4*a*c;
            if(delta<0){
                x1=x2=0.0;
                return 0;
            }
            else if(delta==0){
                x1 = x2 = -b/(2*a);
                return 1;
            }
            else{
                delta = sqrt(delta);
                x1 = (-b + delta) / (2*a);
                x2 = (-b - delta) / (2*a);
                return 2;
            }
        }
