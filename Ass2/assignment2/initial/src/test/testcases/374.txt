
        int numberDays(int month,int year)
        { 

            if (year> MAX_YR || year < MIN_YR)  return 0;
	        if (month< 1 || month > 12) return 0;
            if (month == 2)
            {
                if (IsLeapYear(year)) return (29);
                else
                    return (28);
            }
	        else if (month == 4 || month == 6 || month == 9 || month== 11)
                        return (30);
		
            return 31;
        }
