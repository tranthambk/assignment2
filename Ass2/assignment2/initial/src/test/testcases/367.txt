        void insertKey(int k) 
        { 
	        if (heap_size == capacity) return; 

	        heap_size=heap_size+1; 
	        int i ;i= heap_size - 1; 
	        harr[i] = k; 

	        do
	        { 
	            swap(harr[i], harr[parent(i)]); 
	            i = parent(i); 
	        } while(i != 0 && harr[parent(i)] > harr[i]) ;
        }
