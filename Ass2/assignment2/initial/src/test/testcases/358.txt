
        void heapSort(int arr[], int n) 
        { 
            int i;
            for (i = n / 2 - 1; i >= 0; i=i-1) 
                    heapify(arr, n, i); 
            for (i=n-1; i>=0; i=i-1) 
            { 
                swap(arr[0], arr[i]); 
                heapify(arr, i, 0); 
            } 
        }
