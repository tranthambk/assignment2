
        void merge(int arr[], int l, int m, int r) 
        { 
            int i, j, k; int n1 ;n1= m - l + 1; 
            int n2 ;n2=  r - m; 
  
            int L[100], R[100]; 
            for (i = 0; i < n1; i=i+1) L[i] = arr[l + i]; 
            for (j = 0; j < n2; j=j+1) R[j] = arr[m + 1+ j]; 
            i = 0; j = 0; k = l; 
            do
            { 
                if (L[i] <= R[j]) 
                { 
                    arr[k] = L[i]; 
                    i=i+1; 
                } 
                else
                { 
                    arr[k] = R[j]; 
                    j=j+1; 
                } 
                k=k+1; 
            } while (i < n1 && j < n2) ;
            do
            { 
                arr[k] = L[i]; 
                i=i+1; 
                k=k+1; 
            } while (i < n1) ;
            do
            { 
                arr[k] = R[j]; 
                j=j+1; 
                k=k+1; 
            } while (j < n2) ;
        }
