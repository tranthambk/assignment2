
        float F(float x,int n) {
            int a,i;
            if (n % 2 == 0)
                a = 1;
            else
                a = -1;
            float temp ;temp= 1;
            for (i = 1; i <= 2 * n + 1; i=i+1)
                temp = temp * x / i;
            return a * temp;
        }
