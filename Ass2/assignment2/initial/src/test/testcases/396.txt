        boolean isPrime(int num)
        {
            if((num <= 1) || (num % 2 ==0 && num != 2))
                return false;
            int square, i;
            square = sqrt(num);
            for (i=2;i<=square+1;i = i+1)
             {
                if (num % i == 0) return false;
             }
            return true;
        }
